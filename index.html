function updateDateTime() { var now = new Date(); var dateTimeString = now.getFullYear() + '年' + (now.getMonth() + 1).toString().padStart(2, '0') + '月' + now.getDate().toString().padStart(2, '0') + '日 ' + now.getHours().toString().padStart(2, '0') + ':' + now.getMinutes().toString().padStart(2, '0') + ':' + now.getSeconds().toString().padStart(2, '0'); document.getElementById('datetime').textContent = dateTimeString; } updateDateTime(); setInterval(updateDateTime, 1000);
